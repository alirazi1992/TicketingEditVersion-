         } else {
           // Not a supervisor, clear the list
           setAvailableTechnicians([]);
         }
       } catch (error: any) {
         console.error("Failed to load technician profile:", error);
         // Don't crash the dashboard - just log the error
         if (error?.status === 403) {
           console.warn("Access denied to technician profile endpoint");
         }
       } finally {
         loadingProfileRef.current = false;
       }
     };
     
     loadTechnicianProfileAndCheckSupervisor();
   }, [token, currentUser?.id]);
